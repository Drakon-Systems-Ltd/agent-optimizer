## Phase 2 ‚Äî Detection module

### Task 2: `src/detect/index.ts`

**Files:**
- Create: `src/detect/index.ts`
- Create: `tests/detect.test.ts`

Implementation:

```typescript
import { existsSync, readFileSync } from "fs";
import { homedir } from "os";
import { resolve } from "path";

export type SystemKind = "claude-code" | "openclaw" | "cursor";

export interface DetectedSystem {
  kind: SystemKind;
  version: string | null;
  configPath: string;
  scope: "user" | "project";
}

export function detectSystems(cwd: string = process.cwd()): DetectedSystem[] {
  const found: DetectedSystem[] = [];

  // Claude Code ‚Äî user
  const ccUser = resolve(homedir(), ".claude", "settings.json");
  if (existsSync(ccUser)) {
    found.push({ kind: "claude-code", version: detectClaudeCodeVersion(), configPath: ccUser, scope: "user" });
  }

  // Claude Code ‚Äî project
  const ccProjSettings = resolve(cwd, ".claude", "settings.json");
  const ccProjMd = resolve(cwd, "CLAUDE.md");
  if (existsSync(ccProjSettings) || existsSync(ccProjMd)) {
    found.push({
      kind: "claude-code",
      version: null,
      configPath: existsSync(ccProjSettings) ? ccProjSettings : ccProjMd,
      scope: "project",
    });
  }

  // OpenClaw
  const oc = resolve(homedir(), ".openclaw", "openclaw.json");
  if (existsSync(oc)) {
    found.push({ kind: "openclaw", version: detectOpenClawVersion(), configPath: oc, scope: "user" });
  }

  // Cursor (project-scope only for now)
  const cursor = resolve(cwd, ".cursor", "rules");
  if (existsSync(cursor)) {
    found.push({ kind: "cursor", version: null, configPath: cursor, scope: "project" });
  }

  return found;
}

function detectClaudeCodeVersion(): string | null {
  try {
    const { execSync } = require("child_process");
    const v = execSync("claude --version 2>/dev/null", { timeout: 3000, encoding: "utf-8" }).trim();
    const m = v.match(/(\d+\.\d+\.\d+)/);
    return m ? m[1] : null;
  } catch {
    return null;
  }
}
```

```typescript
program
  .command("detect")
  .description("Detect installed Claude-family agent systems")
  .option("--json", "JSON output")
  .action((opts) => {
    const systems = detectSystems();
    if (opts.json) {
      console.log(JSON.stringify(systems, null, 2));
      return;
    }
    if (systems.length === 0) {
      console.log("No Claude-family systems detected.");
      return;
    }
    console.log(chalk.bold("Detected systems:"));
    for (const s of systems) {
      console.log(`  ${chalk.cyan(s.kind)} ${s.version ?? ""} (${s.scope}) ‚Äî ${s.configPath}`);

  });
```
